28//! ## Motivation
29//!
30//! I have a few Rust projects that are compiled for Linux and Windows. I need to save some paths inside a config file. Windows have a strange way to work with file/folder paths. I need a library to work in a neutral crossplatform way. Only at the last line of code I transform the neutral path into something that the current OS recognizes.
31//!
32//! There exist already some libraries for that: relative-path, typed-path, x-path, camino,...
33//!
34//! But it is fun to make something new and simple and very very opinionated.
35//!
36//! ## Opinionated to the max
37//!
38//! My opinions are probably not useful for all developers, but they work for me and probably for most.
39//!
40//! 1. The path will be strictly utf8. I know that there exists a rare possibility of the path being in some other strange format, but I will never encounter that with this library. Or at least, I will always avoid that.
41//! 2. Unix and Linux have paths that look nice. Windows is the problem here. The crossplatform format will be very much like the Linux paths.
42//! 3. The only path separator will be the universal '/'. If some '\\' exists, it will be replaced by '/'. Linux allows the use of '\\' inside a filename, but my opinion is that this is bad and should be avoided.
43//! 4. Linux is very permissive. Only NULL and '/' are forbidden in path components. But it is a good idea to not allow special characters forbidden on Windows:  
44//!
45//!     ```text
46//!     < > : " / \\ | ? *
47//!     0 (NULL byte)
48//!     0-31 (ASCII control characters)  
49//!     ```
50//!   
51//! 5. Filenames cannot end in a space or dot.
52//! 6. Not allow reserved filenames even with extensions and foldernames:  
53//!    CON, PRN, AUX, NUL  
54//!    COM1, COM2, COM3, COM4, COM5, COM6, COM7, COM8, COM9  
55//!    LPT1, LPT2, LPT3, LPT4, LPT5, LPT6, LPT7, LPT8, LPT9  
56//!    These names are not really needed and will not be allowed:  
57//!    .  (special name referring to current directory)  
58//!    This have to be avoided because of traversal attacks:  
59//!    .. (special name referring to parent directory)  
60//!
61//! 7. Instead of the problematic Windows 'c:' or 'd:' drives,  
62//!    the neutral crossplatform format will be '/mnt/c' or '/mnt/d'  
63//!    From Windows:  
64//!    c:\\ will be transformed into /mnt/c/  
65//!    d:\\ will be transformed into /mnt/d/  
66//! 8. This special symbols and root folders are allowed and will be transformed for Windows:  
67//!    '~'    will be transformed into %UserProfile%  
68//!    /tmp   will be transformed into %TEMP%  
69//! 9. Definitely some paths in one OS have absolutely no meaning in other OS, but these have to be avoided manually.
70//!
71//! ## Usage
72//!
73//! ```rust
74//! // cargo add crossplatform_path
75//!
76//! let cross_path = crossplatform_path::CrossPathBuf::new(r#"tmp\folder_1"#)?.join_relative(r#"file_1.txt"#)?;

79//! let linux_path_buf = cross_path.to_path_buf_nix();

146/// All possible library errors for `thiserror`.
147#[derive(thiserror::Error, Debug)]
148pub enum Error {
149    #[error(r#"The string {0} contains an invalid windows path character < > : " | ? * "#)]
150    InvalidCharacter(String),
151    #[error(r#"The string {0} contains forbidden ascii control character for windows path 0-31 "#)]
152    ForbiddenAscii(String),
153    #[error(r#"The path string {0} must not end with space or dot "#)]
154    MustNotEndWith(String),
155    #[error(r#"The path string {0} must not contain reserved words con, prn, aux, nul, com1-com9, lpt1-lpt9, . and .."#)]
156    ReservedWords(String),
157    #[error(r#"The parent of {0} does not exist."#)]
158    NoParent(String),
159    #[error(r#"The file_name of {0} does not exist."#)]
160    NoFileName(String),
161    #[error("I/O error: {path} {source}")]
162    IoError {
163        #[source]
164        source: std::io::Error,
165        path: String,
166    },
167    #[error("Unknown error.")]
168    Unknown,
169}

178/// CrossPathBuf stores Path in a Neutral Crossplatform format.  \
179///
180/// The neutral path is limited to valid utf8 strings.  \
181/// This format can be stored in config files. It is "similar" to the Linux format, but not exactly equal.  \
182/// When used for file operations, this Neutral format is converted into Linux or Windows format accordingly.  \
183/// Some limitations exist for paths mostly because of Windows limitations:  \
184/// forbidden characters < > : " / \\ | ? *  0 (NULL byte)  0-31 (ASCII control characters)  \
185/// Filenames cannot end in a space or dot.  \
186/// Separator is always slash. Backslash is replaced. Backslash must never be a part of a name or path component.  \
187/// Must not contain reserved words con, prn, aux, nul, com1-com9, lpt1-lpt9, . and ..  \
188/// If starts with windows c: or d: it is converted to /mnt/c or /mnt/d lowercase  
189#[derive(Clone, Debug, PartialEq)]
190pub struct CrossPathBuf {
191    /// Path stored in a Neutral Crossplatform format.
192    cross_path: String,
193}
194
195impl CrossPathBuf {
196    /// Creates a new CrossPathBuf from &str.  \
197    ///
198    /// Path must be always utf8. Rust strings are guaranteed to be utf8.  \
199    /// The input path will be tested that is somehow correct.  \
200    /// It will be transformed from Windows into the crossplatform format. Linux format will stay mostly the same.  \
201    /// The neutral path is limited to valid utf8 strings.  \
202    /// This format can be stored in config files. It is "similar" to the Linux format, but not exactly equal.  \
203    /// When used for file operations, this Neutral format is converted into Linux or Windows format accordingly.  \
204    /// Some limitations exist for paths mostly because of Windows limitations:  \
205    /// forbidden characters < > : " / \\ | ? *  0 (NULL byte)  0-31 (ASCII control characters)  \
206    /// Filenames cannot end in a space or dot.  \
207    /// Separator is always slash. Backslash is replaced. Backslash must never be a part of a name or path component.  \
208    /// Must not contain reserved words con, prn, aux, nul, com1-com9, lpt1-lpt9, . and ..  \
209    /// If start with windows c: or d: convert to /mnt/c or /mnt/d lowercase  
210    pub fn new(str_path: &str) -> Result<Self> {
211        // forbidden: < > : " / \\ | ? *  0 (NULL byte)  0-31 (ASCII control characters)
212        // but : / and \\ are delimiters and can be used in a path fragment with multiple components.
213        if str_path.contains("<")
214            || str_path.contains(">")
215            || str_path.contains(r#"""#)
216            || str_path.contains("|")
217            || str_path.contains("?")
218            || str_path.contains("*")
219        {
220            return Err(Error::InvalidCharacter(str_path.to_string()));
221        }
222        // 0 (NULL byte) and  0-31 (ASCII control characters) 127 is DEL
223        // Important: utf8 is always on a byte level compatible with ASCII7, under 127.
224        for byte in str_path.bytes() {
225            match byte {
226                0x00..=0x1F | 0x7F => return Err(Error::ForbiddenAscii(str_path.to_string())),
227                _ => (),
228            }
229        }
230        //Filenames cannot end in a space or dot.
231        if str_path.ends_with(" ") || str_path.ends_with(".") {
232            return Err(Error::MustNotEndWith(str_path.to_string()));
233        }
234
235        // Separator is always slash. Backslash is replaced. Backslash must never be a part of a name or path component.
236        // trim: leading and trailing whitespace removed
237        let mut cross_path = str_path.trim().replace(r#"\"#, "/");
238
239        // 6. Not allow reserved filenames even with extensions and foldernames:
240        // Windows path is case insensitive, so I must check insensitive. I will use to_lowercase.
241        // CON, PRN, AUX, NUL
242        // COM1, COM2, COM3, COM4, COM5, COM6, COM7, COM8, COM9
243        // LPT1, LPT2, LPT3, LPT4, LPT5, LPT6, LPT7, LPT8, LPT9
244        // These names are not really needed and will not be allowed:
245        // .  (special name referring to current directory)
246        // This have to be avoided because of traversal attacks:
247        // .. (special name referring to parent directory)
248        // I need to delimit start and end with / to use contains.
249        let delimited_str_path = format!("/{}/", cross_path.trim_start_matches("/").trim_end_matches("/").to_lowercase());
250        if delimited_str_path.contains("/con/")
251            || delimited_str_path.contains("/prn/")
252            || delimited_str_path.contains("/aux/")
253            || delimited_str_path.contains("/nul/")
254            || delimited_str_path.contains("/com1/")
255            || delimited_str_path.contains("/com2/")
256            || delimited_str_path.contains("/com3/")
257            || delimited_str_path.contains("/com4/")
258            || delimited_str_path.contains("/com5/")
259            || delimited_str_path.contains("/com6/")
260            || delimited_str_path.contains("/com7/")
261            || delimited_str_path.contains("/com8/")
262            || delimited_str_path.contains("/com9/")
263            || delimited_str_path.contains("/lpt1/")
264            || delimited_str_path.contains("/lpt2/")
265            || delimited_str_path.contains("/lpt3/")
266            || delimited_str_path.contains("/lpt4/")
267            || delimited_str_path.contains("/lpt5/")
268            || delimited_str_path.contains("/lpt6/")
269            || delimited_str_path.contains("/lpt7/")
270            || delimited_str_path.contains("/lpt8/")
271            || delimited_str_path.contains("/lpt9/")
272            || delimited_str_path.contains("/./")
273            || delimited_str_path.contains("/../")
274        {
275            return Err(Error::ReservedWords(str_path.to_string()));
276        }
277
278        // If start with windows c: or d: convert to /mnt/c or /mnt/d lowercase
279        let mut iter = cross_path.chars();
280        if let Some(first) = iter.next()
281            && let Some(second) = iter.next()
282            && second == ':'
283        {
284            cross_path = format!("/mnt/{}{}", first.to_lowercase(), iter.as_str());
285        }
286
287        // Forbidden character, except for windows drive
288        if cross_path.contains(":") {
289            return Err(Error::InvalidCharacter(cross_path));
290        }
291
292        // Forbidden double slash
293        if cross_path.contains("//") {
294            return Err(Error::InvalidCharacter(cross_path));
295        }
296
297        Ok(CrossPathBuf { cross_path })
298    }

300    /// Converts crossplatform path into Windows path.  \
301    ///
302    /// '~'     will be transformed into home  \
303    /// /mnt/c/ will be transformed into c:\\  \
304    /// /mnt/d/ will be transformed into d:\\  \
305    /// /tmp    will be transformed into %TEMP%  
306    pub fn to_path_buf_win(&self) -> std::path::PathBuf {
307        let mut win_path = self.cross_path.clone();
308        // '~'    will be transformed into home
309        if win_path.starts_with("~")
310            && let Some(home) = std::env::home_dir()
311        {
312            win_path = format!("{}{}", home.to_string_lossy(), win_path.trim_start_matches("~"));
313        }
314        // /mnt/c/ will be transformed into c:\\
315        // /mnt/d/ will be transformed into d:\\
316        if win_path.starts_with("/mnt/") {
317            win_path = win_path.trim_start_matches("/mnt/").to_string();
318            win_path.insert(1, ':');
319        }
320        // /tmp   will be transformed into %TEMP%
321        if win_path.starts_with("/tmp") {
322            let tmp_dir = std::env::temp_dir();
323            win_path = format!("{}{}", tmp_dir.to_string_lossy(), win_path.trim_start_matches("/tmp"));
324        }
325        use std::str::FromStr;
326        std::path::PathBuf::from_str(&win_path).expect("PathBuf::from_str() returns Infallible error. Therefore the error cannot occur.")
327    }

329    /// Converts crossplatform path into Linux path.  \
330    ///
331    /// '~'    will be transformed into home  
332    pub fn to_path_buf_nix(&self) -> std::path::PathBuf {
333        let mut nix_path = self.cross_path.clone();
334        // '~'    will be transformed into home
335        if nix_path.starts_with("~")
336            && let Some(home) = std::env::home_dir()
337        {
338            nix_path = format!("{}{}", home.to_string_lossy(), nix_path.trim_start_matches("~"));
339        }
340        use std::str::FromStr;
341        std::path::PathBuf::from_str(&nix_path).expect("PathBuf::from_str() returns Infallible error. Therefore the error cannot occur.")
342    }

344    /// Converts crossplatform path into current OS path.  \
345    ///
346    /// '~'     will be transformed into home  \
347    /// /mnt/c/ will be transformed into c:\\  \
348    /// /mnt/d/ will be transformed into d:\\  \
349    /// /tmp    will be transformed into %TEMP%  
350    pub fn to_path_buf_current_os(&self) -> std::path::PathBuf {
351        if cfg!(windows) {
352            self.to_path_buf_win()
353        } else {
354            self.to_path_buf_nix()
355        }
356    }

390    /// Joins two paths and returns a new CrossPathBuf to allow function chaining.  \
391    ///
392    /// It works differently from the original Rust join() where if the second path is absolute, it overwrites the first path.  \
393    /// Here the second path is always relative and is added to the first path.
394    pub fn join_relative(&self, str_path: &str) -> Result<Self> {
395        let second_path = CrossPathBuf::new(str_path)?;
396        let cross_path = format!(
397            "{}/{}",
398            self.cross_path.trim_end_matches("/"),
399            second_path.as_str().trim_start_matches("/")
400        );
401        Ok(CrossPathBuf { cross_path })
402    }

404    /// Reads the entire contents of a file into a string.  \
405    ///
406    /// This is a convenience function based on std::fs::read_to_string  
407    pub fn read_to_string(&self) -> Result<String> {
408        match std::fs::read_to_string(self.to_path_buf_current_os()) {
409            Ok(content) => Ok(content),
410            Err(err) => Err(Error::IoError {
411                source: err,
412                path: self.cross_path.to_string(),
413            }),
414        }
415    }

417    /// Writes a slice as the entire contents of a file.  \
418    ///
419    /// This function will create a file if it does not exist, and will entirely replace its contents if it does.  \
420    /// It creates the full path directory, if path does not exist.  
421    pub fn write_str_to_file(&self, content: &str) -> Result<()> {
422        self.create_dir_all_for_file()?;
423        match std::fs::write(self.to_path_buf_current_os(), content) {
424            Ok(_) => Ok(()),
425            Err(err) => Err(Error::IoError {
426                source: err,
427                path: self.cross_path.clone(),
428            }),
429        }
430    }

432    /// Recursively create this path as directory and all of its parent components if they are missing.  \
433    ///
434    /// The cross_path must represent a directory and not a file for this command.
435    /// This function is not atomic. If it returns an error, any parent components it was able to create will remain.   
436    pub fn create_dir_all(&self) -> Result<()> {
437        match std::fs::create_dir_all(self.to_path_buf_current_os()) {
438            Ok(_) => Ok(()),
439            Err(err) => Err(Error::IoError {
440                source: err,
441                path: self.cross_path.clone(),
442            }),
443        }
444    }
445
446    /// Recursively create the parent directory of a file and all of its parent components if they are missing.  \
447    ///
448    /// The cross_path must represent a file. The parent directory will be created.
449    /// This function is not atomic. If it returns an error, any parent components it was able to create will remain.   
450    pub fn create_dir_all_for_file(&self) -> Result<()> {
451        let path = self.to_path_buf_current_os();
452        let parent = path.parent().ok_or_else(|| Error::NoParent(self.cross_path.clone()))?;
453        match std::fs::create_dir_all(parent) {
454            Ok(_) => Ok(()),
455            Err(err) => Err(Error::IoError {
456                source: err,
457                path: self.cross_path.to_string(),
458            }),
459        }
460    }

486    /// Returns the final component of the Path, if there is one.
487    ///
488    /// If the path is a normal file, this is the file name.
489    /// If it's the path of a directory, this is the directory name.
490    pub fn file_name(&self) -> Result<String> {
491        let file_name = self
492            .to_path_buf_current_os()
493            .file_name()
494            .ok_or_else(|| Error::NoFileName(self.cross_path.clone()))?
495            .to_string_lossy()
496            .to_string();
497        Ok(file_name)
498    }

500    /// Extracts the extension (without the leading dot), if possible.  
501    pub fn extension(&self) -> Result<String> {
502        let file_name = self
503            .to_path_buf_current_os()
504            .extension()
505            .ok_or_else(|| Error::NoFileName(self.cross_path.clone()))?
506            .to_string_lossy()
507            .to_string();
508        Ok(file_name)
509    }
510
511    /// Extracts the stem (non-extension) portion of file_name (the final component of the Path).
512    pub fn file_stem(&self) -> Result<String> {
513        let file_name = self
514            .to_path_buf_current_os()
515            .file_stem()
516            .ok_or_else(|| Error::NoFileName(self.cross_path.clone()))?
517            .to_string_lossy()
518            .to_string();
519        Ok(file_name)
520    }
521
522    /// Returns the Path without its final component, if there is one.
523    pub fn parent(&self) -> Result<Self> {
524        CrossPathBuf::new(
525            &self
526                .to_path_buf_current_os()
527                .parent()
528                .ok_or_else(|| Error::NoParent(self.cross_path.clone()))?
529                .to_string_lossy(),
530        )
531    }
532}
