26//! ## Motivation
27//!
28//! I have a few Rust projects that are compiled for Linux and Windows. I need to save some paths inside a config file. Windows have a strange way to work with file/folder paths. I need a library to work in a neutral crossplatform way. Only at the last line of code I transform the neutral path into something that the current OS recognizes.
29//!
30//! There exist already some libraries for that: relative-path, typed-path, x-path, camino,...
31//!
32//! But it is fun to make something new and simple and very very opinionated.
33//!
34//! ## Opinionated to the max
35//!
36//! My opinions are probably not useful for all developers, but they work for me and probably for most.
37//!
38//! 1. The path will be strictly utf8. I know that there exists a rare possibility of the path being in some other strange format, but I will never encounter that with this library. Or at least, I will always avoid that.
39//! 2. Unix and Linux have paths that look nice. Windows is the problem here. The crossplatform format will be very much like the Linux paths.
40//! 3. The only path separator will be the universal '/'. If some '\\' exists, it will be replaced by '/'. Linux allows the use of '\\' inside a filename, but my opinion is that this is bad and should be avoided.
41//! 4. Linux is very permissive. Only NULL and '/' are forbidden in path components. But it is a good idea to not allow special characters forbidden on Windows:  
42//!
43//!     ```text
44//!     < > : " / \\ | ? *
45//!     0 (NULL byte)
46//!     0-31 (ASCII control characters)  
47//!     ```
48//!   
49//! 5. Filenames cannot end in a space or dot.
50//! 6. Not allow reserved filenames even with extensions and foldernames:  
51//!    CON, PRN, AUX, NUL  
52//!    COM1, COM2, COM3, COM4, COM5, COM6, COM7, COM8, COM9  
53//!    LPT1, LPT2, LPT3, LPT4, LPT5, LPT6, LPT7, LPT8, LPT9  
54//!    These names are not really needed and will not be allowed:  
55//!    .  (special name referring to current directory)  
56//!    This have to be avoided because of traversal attacks:  
57//!    .. (special name referring to parent directory)  
58//!
59//! 7. Instead of the problematic Windows 'c:' or 'd:' drives,  
60//! the neutral crossplatform format will be '/mnt/c' or '/mnt/d'  
61//!    From Windows:  
62//!    c:\\ will be transformed into /mnt/c/  
63//!    d:\\ will be transformed into /mnt/d/  
64//! 8. This special symbols and root folders are allowed and will be transformed for Windows:  
65//!    '~'    will be transformed into %UserProfile%  
66//!    /tmp   will be transformed into %TEMP%  
67//! 9. Definitely some paths in one OS have absolutely no meaning in other OS, but these have to be avoided manually.

106/// All possible library errors for `thiserror`.
107#[derive(Error, Debug, PartialEq)]
108pub enum LibraryError {
109    #[error(r#"The string {0} contains an invalid windows path character < > : " | ? * "#)]
110    InvalidCharacter(String),
111    #[error(r#"The string {0} contains forbidden ascii control character for windows path 0-31 "#)]
112    ForbiddenAscii(String),
113    #[error(r#"The path string {0} must not end with space or dot "#)]
114    MustNotEndWith(String),
115    #[error(r#"The path string {0} must not contain reserved words con, prn, aux, nul, com1-com9, lpt1-lpt9, . and .."#)]
116    ReservedWords(String),
117    #[error("Unknown error.")]
118    Unknown,
119}

121/// CrossPathBuf stores Path in a Neutral Crossplatform format.
122///
123/// The neutral path is limited to valid utf8 strings.  
124/// This format can be stored in config files. It is "similar" to the Linux format, but not exactly equal.  
125/// When used for file operations, this Neutral format is converted into Linux or Windows format accordingly.  
126/// Some limitations exist for paths mostly because of Windows limitations:  
127/// forbidden characters < > : " / \\ | ? *  0 (NULL byte)  0-31 (ASCII control characters)  
128/// Filenames cannot end in a space or dot.  
129/// Separator is always slash. Backslash is replaced. Backslash must never be a part of a name or path component.  
130/// Must not contain reserved words con, prn, aux, nul, com1-com9, lpt1-lpt9, . and ..  
131/// If starts with windows c: or d: it is converted to /mnt/c or /mnt/d lowercase  
132#[derive(Clone, Debug, PartialEq)]
133pub struct CrossPathBuf {
134    /// Path stored in a Neutral Crossplatform format.
135    cross_path: String,
136}
137
138impl CrossPathBuf {
139    /// Creates a new CrossPathBuf from &str.
140    ///
141    /// Path must be always utf8. Rust strings are guaranteed to be utf8.  
142    /// The input path will be tested that is somehow correct.  
143    /// It will be transformed from Windows into the crossplatform format. Linux format will stay mostly the same.  
144    /// The neutral path is limited to valid utf8 strings.  
145    /// This format can be stored in config files. It is "similar" to the Linux format, but not exactly equal.  
146    /// When used for file operations, this Neutral format is converted into Linux or Windows format accordingly.  
147    /// Some limitations exist for paths mostly because of Windows limitations:  
148    /// forbidden characters < > : " / \\ | ? *  0 (NULL byte)  0-31 (ASCII control characters)  
149    /// Filenames cannot end in a space or dot.  
150    /// Separator is always slash. Backslash is replaced. Backslash must never be a part of a name or path component.  
151    /// Must not contain reserved words con, prn, aux, nul, com1-com9, lpt1-lpt9, . and ..  
152    /// If start with windows c: or d: convert to /mnt/c or /mnt/d lowercase  
153    pub fn new(str_path: &str) -> Result<Self, LibraryError> {
154        // forbidden: < > : " / \\ | ? *  0 (NULL byte)  0-31 (ASCII control characters)
155        // but : / and \\ are delimiters and can be used in a path fragment with multiple components.
156        if str_path.contains("<")
157            || str_path.contains(">")
158            || str_path.contains(r#"""#)
159            || str_path.contains("|")
160            || str_path.contains("?")
161            || str_path.contains("*")
162        {
163            return Err(LibraryError::InvalidCharacter(str_path.to_string()));
164        }
165        // 0 (NULL byte) and  0-31 (ASCII control characters) 127 is DEL
166        // Important: utf8 is always on a byte level compatible with ASCII7, under 127.
167        for byte in str_path.bytes() {
168            match byte {
169                0x00..=0x1F | 0x7F => return Err(LibraryError::ForbiddenAscii(str_path.to_string())),
170                _ => (),
171            }
172        }
173        //Filenames cannot end in a space or dot.
174        if str_path.ends_with(" ") || str_path.ends_with(".") {
175            return Err(LibraryError::MustNotEndWith(str_path.to_string()));
176        }
177
178        // Separator is always slash. Backslash is replaced. Backslash must never be a part of a name or path component.
179        let mut cross_path = str_path.replace(r#"\"#, "/");
180
181        // 6. Not allow reserved filenames even with extensions and foldernames:
182        // Windows path is case insensitive, so I must check insensitive. I will use to_lowercase.
183        // CON, PRN, AUX, NUL
184        // COM1, COM2, COM3, COM4, COM5, COM6, COM7, COM8, COM9
185        // LPT1, LPT2, LPT3, LPT4, LPT5, LPT6, LPT7, LPT8, LPT9
186        // These names are not really needed and will not be allowed:
187        // .  (special name referring to current directory)
188        // This have to be avoided because of traversal attacks:
189        // .. (special name referring to parent directory)
190        // I need to delimit start and end with / to use contains.
191        let delimited_str_path = format!("/{}/", cross_path.trim_start_matches("/").trim_end_matches("/").to_lowercase());
192        if delimited_str_path.contains("/con/")
193            || delimited_str_path.contains("/prn/")
194            || delimited_str_path.contains("/aux/")
195            || delimited_str_path.contains("/nul/")
196            || delimited_str_path.contains("/com1/")
197            || delimited_str_path.contains("/com2/")
198            || delimited_str_path.contains("/com3/")
199            || delimited_str_path.contains("/com4/")
200            || delimited_str_path.contains("/com5/")
201            || delimited_str_path.contains("/com6/")
202            || delimited_str_path.contains("/com7/")
203            || delimited_str_path.contains("/com8/")
204            || delimited_str_path.contains("/com9/")
205            || delimited_str_path.contains("/lpt1/")
206            || delimited_str_path.contains("/lpt2/")
207            || delimited_str_path.contains("/lpt3/")
208            || delimited_str_path.contains("/lpt4/")
209            || delimited_str_path.contains("/lpt5/")
210            || delimited_str_path.contains("/lpt6/")
211            || delimited_str_path.contains("/lpt7/")
212            || delimited_str_path.contains("/lpt8/")
213            || delimited_str_path.contains("/lpt9/")
214            || delimited_str_path.contains("/./")
215            || delimited_str_path.contains("/../")
216        {
217            return Err(LibraryError::ReservedWords(str_path.to_string()));
218        }
219
220        // If start with windows c: or d: convert to /mnt/c or /mnt/d lowercase
221        let mut iter = cross_path.chars();
222        if let Some(first) = iter.next()
223            && let Some(second) = iter.next()
224            && second == ':'
225        {
226            cross_path = format!("/mnt/{}{}", first.to_lowercase(), iter.as_str());
227        }
228
229        // Forbidden character, except for windows drive
230        if cross_path.contains(":") {
231            return Err(LibraryError::InvalidCharacter(cross_path));
232        }
233
234        // Forbidden double slash
235        if cross_path.contains("//") {
236            return Err(LibraryError::InvalidCharacter(cross_path));
237        }
238
239        Ok(CrossPathBuf { cross_path })
240    }

242    /// Converts crossplatform path into Windows path.
243    ///
244    /// '~'     will be transformed into home  
245    /// /mnt/c/ will be transformed into c:\\  
246    /// /mnt/d/ will be transformed into d:\\  
247    /// /tmp    will be transformed into %TEMP%  
248    pub fn to_win_path_buf(&self) -> std::path::PathBuf {
249        let mut win_path = self.cross_path.clone();
250        // '~'    will be transformed into home
251        if win_path.starts_with("~")
252            && let Some(home) = std::env::home_dir()
253        {
254            win_path = format!("{}{}", home.to_string_lossy(), win_path.trim_start_matches("~"));
255        }
256        // /mnt/c/ will be transformed into c:\\
257        // /mnt/d/ will be transformed into d:\\
258        if win_path.starts_with("/mnt/") {
259            win_path = win_path.trim_start_matches("/mnt/").to_string();
260            win_path.insert(1, ':');
261        }
262        // /tmp   will be transformed into %TEMP%
263        if win_path.starts_with("/tmp") {
264            let tmp_dir = std::env::temp_dir();
265            win_path = format!("{}{}", tmp_dir.to_string_lossy(), win_path.trim_start_matches("/tmp"));
266        }
267        use std::str::FromStr;
268        std::path::PathBuf::from_str(&win_path).expect("PathBuf::from_str() returns Infallible error. Therefore the error cannot occur.")
269    }
270
271    /// Converts crossplatform path into Linux path.
272    ///
273    /// '~'    will be transformed into home  
274    pub fn to_nix_path_buf(&self) -> std::path::PathBuf {
275        let mut nix_path = self.cross_path.clone();
276        // '~'    will be transformed into home
277        if nix_path.starts_with("~")
278            && let Some(home) = std::env::home_dir()
279        {
280            nix_path = format!("{}{}", home.to_string_lossy(), nix_path.trim_start_matches("~"));
281        }
282        use std::str::FromStr;
283        std::path::PathBuf::from_str(&nix_path).expect("PathBuf::from_str() returns Infallible error. Therefore the error cannot occur.")
284    }

286    /// Converts crossplatform path into current OS path.
287    ///
288    /// '~'     will be transformed into home  
289    /// /mnt/c/ will be transformed into c:\\  
290    /// /mnt/d/ will be transformed into d:\\  
291    /// /tmp    will be transformed into %TEMP%  
292    pub fn to_current_os_path_buf(&self) -> std::path::PathBuf {
293        if cfg!(windows) {
294            self.to_win_path_buf()
295        } else {
296            self.to_nix_path_buf()
297        }
298    }
299
300    /// Returns the crossplatform str for use in Display and store into config files.
301    pub fn as_str(&self) -> &str {
302        &self.cross_path
303    }
304
305    /// Returns `true` if the path points at an existing entity.
306    pub fn exists(&self) -> bool {
307        if cfg!(windows) {
308            self.to_win_path_buf().exists()
309        } else {
310            self.to_nix_path_buf().exists()
311        }
312    }
313
314    /// Returns `true` if the path exists on disk and is pointing at a regular file.
315    pub fn is_file(&self) -> bool {
316        if cfg!(windows) {
317            self.to_win_path_buf().is_file()
318        } else {
319            self.to_nix_path_buf().is_file()
320        }
321    }
322
323    /// Returns `true` if the path exists on disk and is pointing at a directory.
324    pub fn is_dir(&self) -> bool {
325        if cfg!(windows) {
326            self.to_win_path_buf().is_dir()
327        } else {
328            self.to_nix_path_buf().is_dir()
329        }
330    }
331
332    /// Joins two paths and returns a new CrossPathBuf to allow function chaining.
333    ///
334    /// It works differently from the original Rust join() where if the second path is absolute, it overwrites the first path.  
335    /// Here the second path is always relative and is added to the first path.  
336    pub fn join_relative(&self, str_path: &str) -> Result<Self, LibraryError> {
337        let second_path = CrossPathBuf::new(str_path)?;
338        let cross_path = format!(
339            "{}/{}",
340            self.cross_path.trim_end_matches("/"),
341            second_path.as_str().trim_start_matches("/")
342        );
343        Ok(CrossPathBuf { cross_path })
344    }
345}

355/// CrossPathBuf from() and into() are useful in places where PathBuf is needed.
356impl From<CrossPathBuf> for std::path::PathBuf {
357    /// CrossPathBuf from() and into() are useful in places where PathBuf is needed.
358    fn from(cross_path: CrossPathBuf) -> Self {

360    }
